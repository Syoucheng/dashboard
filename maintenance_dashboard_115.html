🔧 第一次使用：需建立 Google Apps Script 橋接器（只需做一次）
  1. 開啟你的 Google Sheets → 上方選單「擴充功能」→「Apps Script」
  2. 刪除預設程式碼，複製下方框框內的程式碼貼上
  3. 點上方「儲存」→「部署」→「新增部署作業」
  4. 類型選「網頁應用程式」，執行身分選「我」，存取權選「任何人」
  5. 點「部署」→「授權存取」→ 複製產生的 網頁應用程式 URL
  6. 將 URL 貼到下方「Apps Script 網址」欄位，點「儲存並載入」
📋 Apps Script 程式碼（複製後貼到 Apps Script 編輯器） 
function doGet(e) {
  var sheetId   = e.parameter.id;
  var sheetName = e.parameter.sheet;
  try {
    var ss  = SpreadsheetApp.openById(sheetId);
    var ws  = ss.getSheetByName(sheetName);
    if (!ws) return out({error: "找不到工作表: " + sheetName});
    var data = ws.getDataRange().getValues();
    return out(data);
  } catch(err) {
    return out({error: err.toString()});
  }
}
function out(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
