Google Sheets 雲端設定 請完成設定後載入資料
🔧 第一次使用：需建立 Apps Script 橋接器（只需做一次）
  1. Google Sheets → 上方「擴充功能」→「Apps Script」
  2. 點右上角「部署」→「新增部署作業」→ 類型：網頁應用程式
  3. 存取權選「所有人」→ 部署 → 授權 → 複製 URL
  4. 貼到下方 Apps Script 網址欄位，點「儲存並載入」
📋 Apps Script 程式碼 
// ── 固定設定 ─────────────────────────────────
var SHEET_ID = "1B4GZC9rZx-Ryt-TKlD6_dUDaordo_KhABKadAGDokDI";
var SHEETS   = ["C3EM機電系統-佛廣大","C3T交控系統-華電","C3EM機電系統-中興電工"];
var CACHE_SEC = 360;

function doGet(e) {
  var sheetName = e.parameter.sheet;
  var cache = CacheService.getScriptCache();
  var cacheKey = "sheet_" + sheetName;
  var cached = cache.get(cacheKey);
  if (cached) return out(cached);
  try {
    var ss = SpreadsheetApp.openById(SHEET_ID);
    var ws = ss.getSheetByName(sheetName);
    if (!ws) return out(JSON.stringify({error:"找不到工作表:"+sheetName}));
    var json = JSON.stringify(ws.getDataRange().getValues());
    try { cache.put(cacheKey, json, CACHE_SEC); } catch(e){}
    return out(json);
  } catch(err) { return out(JSON.stringify({error:err.toString()})); }
}

function onSheetChange(e) {
  CacheService.getScriptCache().removeAll(SHEETS.map(s=>"sheet_"+s));
}

function warmCache() {
  var cache = CacheService.getScriptCache();
  var ss = SpreadsheetApp.openById(SHEET_ID);
  SHEETS.forEach(function(s){
    try{
      var json = JSON.stringify(ss.getSheetByName(s).getDataRange().getValues());
      cache.put("sheet_"+s, json, CACHE_SEC);
    }catch(e){}
  });
}

function setupTriggers() {
  ScriptApp.getProjectTriggers().forEach(t=>ScriptApp.deleteTrigger(t));
  var ss = SpreadsheetApp.openById(SHEET_ID);
  ScriptApp.newTrigger("onSheetChange").forSpreadsheet(ss).onChange().create();
  ScriptApp.newTrigger("warmCache").timeBased().everyMinutes(5).create();
}

function out(j) {
  return ContentService.createTextOutput(j).setMimeType(ContentService.MimeType.JSON);
}
